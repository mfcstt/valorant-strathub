# Configurações do Supabase (Exemplo)
# Copie este conteúdo para o seu arquivo .env e preencha com suas credenciais reais.
# NÃO faça commit de chaves reais no repositório.

# Banco de dados Supabase (PostgreSQL)
DB_HOST=YOUR_DB_HOST
DB_PORT=5432
DB_NAME=postgres
DB_USER=postgres
DB_PASSWORD=YOUR_DB_PASSWORD
# DATABASE_URL=postgresql://postgres:YOUR_DB_PASSWORD@YOUR_DB_HOST:5432/postgres

# Supabase Storage e autenticação
SUPABASE_URL=https://YOUR_PROJECT_REF.supabase.co
SUPABASE_ANON_KEY=YOUR_ANON_PUBLIC_KEY
SUPABASE_SERVICE_KEY=YOUR_SERVICE_ROLE_SECRET_KEY

# SQLite local (opcional)
# Defina como 'true' para usar SQLite local ao invés de Postgres
USE_SQLITE=false
